Un ejemplo completo con algunas funciones públicas que pueden ser accedidas desde cualquier elemento que lo soporte.

WebService.h

#pragma once

 

#using "System.EnterpriseServices.dll"

 

//---------------------------------------------------------------------------------------------------------------------

//-- es mejor usar namespaces para las cosas web, por que podremos agrupar varias clases juntas a la hora de invocarlas

namespace nsWebService

{

  //-- Estructuras

       //---------------------------------------------------------------------------------------------------------------------

  public ref class cClienteData

       {

  public:

    cClienteData()

    {

      m_strDato     = "";

      m_strLetraDNI = "";

      m_nDNI        = 0;

    }

 

    cClienteData(cClienteData ^pClienteData)

    {

      m_strDato     = pClienteData->m_strDato;

      m_strLetraDNI = pClienteData->m_strLetraDNI;

      m_nDNI        = pClienteData->m_nDNI;

    }

 

             String ^m_strDato;

    String ^m_strLetraDNI;

             int    m_nDNI;

       };

 

  //---------------------------------------------------------------------------------------------------------------------

       typedef array<cClienteData^> ^stListCliente;

 

  //-- Métodos público Web

  //---------------------------------------------------------------------------------------------------------------------

  //---------------------------------------------------------------------------------------------------------------------

       ref class cWebService

       {

  protected:

   

       public:

             [WebMethod] int           GetInt_Edad();

    [WebMethod] String        ^GetString_Nombre();

             [WebMethod] cClienteData  ^GetClienteData_MisDatos();

             [WebMethod] stListCliente GetListCliente_ListaClientes(int nDNI);

       };

}

WebService.cpp

#using <mscorlib.dll>

#using "System.Web.dll"

#using "System.Web.Services.dll"

 

using namespace System;

using namespace System::Web;

using namespace System::Web::Services;

 

#include "PrestoWebService.h"

 

//---------------------------------------------------------------------------------------------------------------------

namespace nsWebService

{

  //---------------------------------------------------------------------------------------------------------------------

  //-- Devolviendo un entero

       int cWebService::GetInt_Edad()

       {

             return 39;

       }

 

  //---------------------------------------------------------------------------------------------------------------------

  //-- Devolviendo un entero

  String ^ cWebService::GetString_Nombre()

  {

    return "Enrique";

  }

 

  //---------------------------------------------------------------------------------------------------------------------

  //-- Devolviendo una estructura de datos

       cClienteData ^cWebService::GetClienteData_MisDatos()

       {

             cClienteData ^data = gcnew cClienteData();

             data->m_strDato = gcnew String("Enrique");

             data->m_nDNI = 1234567;

    data->m_strLetraDNI = gcnew String("G");

 

             return data;

       }

 

  //---------------------------------------------------------------------------------------------------------------------

  //-- Devolviendo una lista de datos

       stListCliente cWebService::GetListCliente_ListaClientes(int nNumber)

       {

    stListCliente data = gcnew array<cClienteData^>(nNumber);

   

    for each(cClienteData^% i in data)

    {

      i = gcnew cClienteData();

      i->m_strDato = gcnew String("Enrique S.");

      i->m_nDNI = nNumber;

      i->m_strLetraDNI = "L";

    }

   

             return data;

       }

};
